{{/* Render images or videos from local or remote sources. - Resize local images and builds responsive set. - Add popover lightbox. */}} {{- $url := urls.Parse .Destination -}} {{- $sizes := slice "500" "1000" -}} {{- $format := "webp" -}} {{- $id := printf "img-%s" (delimit (shuffle (seq 1 9)) "") -}} {{- $alt := .Text | default (i18n "image_fallback" | default "Image") -}} {{- $title := .Title | plainify -}} {{- $video_fallback := i18n "video_fallback" | default "A video should have been here, but your browser does not support it." -}} {{- $zoom := i18n "zoom" | default "Zoom image" -}} {{- $close := i18n "close" | default "Close image viewer" -}} {{- if $url.Scheme -}} {{/* Remote image */}} {{- with try (resources.GetRemote .Destination) -}} {{- with .Err -}} {{- errorf "%s" . -}} {{- else with .Value -}} {{- end -}} {{- else -}} {{- errorf "Unable to get remote resource %q" .Destination -}} {{- end -}} {{- else -}} {{/* Local (bundled/static) image or video */}} {{- $img := .Page.Resources.GetMatch .Destination | default (resources.Get .Destination) -}} {{- if $img -}} {{- if eq $img.MediaType.MainType "video" -}} {{- else if not (in (slice "jpeg" "jpg" "png" "webp" "gif") $img.MediaType.SubType) -}} {{/* SVG, AVIF, TIFF, BMP, ICO cannot be resized or processed */}} {{- else -}} {{/* Responsive image with popover */}}
{{- end -}} {{- else -}} {{- warnf "No image found for %q in page %q" .Destination .Page.File -}} {{- end -}} {{- end -}}
